## Overview

The Reservation Management module is the core booking system that connects clients with rooms. Staff can create new reservations, update booking details, and manage the complete reservation lifecycle.

<Note>
  Reservation data is managed through the API endpoint `http://localhost:4000/reservaciones`
</Note>

## User Interface

The reservations interface (`reservaciones.html`) displays a streamlined table view with:

- **Folio**: Unique reservation reference number
- **Cliente**: Client name associated with the reservation
- **Acciones**: Action buttons for updating or deleting reservations

## API Implementation

The reservation system implements complete CRUD operations:

<Tabs>
  <Tab title="Create Reservation">
    ```javascript
    async function guardarReservacion() {
        const folio = document.getElementById("folioReservacion").value;
        const nombrecliente = document.getElementById("clienteReservacion").value;

        const data = { folio, nombrecliente };

        try {
            const response = await fetch(URL_RESERVACIONES, {
                method: "POST",
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data)
            });

            if (!response.ok) throw new Error('Error al guardar reservación');

            loadReservaciones();
            limpiarFormulario('formDataReservacion');

        } catch (error) {
            console.error('Error al guardar reservación:', error);
        }
    }
    ```

    **Endpoint**: `POST /reservaciones`

    **Request Body**:
    ```json
    {
      "folio": "RES-2024-001",
      "nombrecliente": "John Doe"
    }
    ```

    After successful creation, the form clears and the reservation list refreshes.
  </Tab>

  <Tab title="Read Reservations">
    ```javascript
    async function loadReservaciones() {
        try {
            const response = await fetch(URL_RESERVACIONES);
            reservaciones = await response.json();

            let html = '';
            reservaciones.forEach(reservacion => {
                html += `
                <tr>
                    <td>${reservacion.folio}</td>
                    <td>${reservacion.nombrecliente}</td>
                    <td>
                        <button class="btn btn-primary btn-sm" 
                                data-bs-toggle="modal" 
                                data-bs-target="#updateReservacion" 
                                onclick="seleccionarReservacion(${reservacion.id})">
                            Actualizar
                        </button>
                        <button class="btn btn-danger btn-sm" 
                                onclick="eliminarReservacion(${reservacion.id})">
                            Eliminar
                        </button>
                    </td>
                </tr>`;
            });

            T_DATA_RESERVACIONES.innerHTML = html;

        } catch (error) {
            console.error('Error al cargar las reservaciones:', error);
        }
    }
    ```

    **Endpoint**: `GET /reservaciones`

    **Response**:
    ```json
    [
      {
        "id": 1,
        "folio": "RES-2024-001",
        "nombrecliente": "John Doe"
      },
      {
        "id": 2,
        "folio": "RES-2024-002",
        "nombrecliente": "Jane Smith"
      }
    ]
    ```
  </Tab>

  <Tab title="Update Reservation">
    ```javascript
    function seleccionarReservacion(id) {
        reservacionSeleccionada = reservaciones.find(res => res.id === id);
        document.getElementById("folioReservacionU").value = reservacionSeleccionada.folio;
        document.getElementById("clienteReservacionU").value = reservacionSeleccionada.nombrecliente;
    }

    async function actualizarReservacion() {
        const folio = document.getElementById("folioReservacionU").value;
        const nombrecliente = document.getElementById("clienteReservacionU").value;

        const data = { folio, nombrecliente };

        try {
            const response = await fetch(`${URL_RESERVACIONES}/${reservacionSeleccionada.id}`, {
                method: "PUT",
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data)
            });

            if (!response.ok) throw new Error('Error al actualizar reservación');

            loadReservaciones();
            limpiarFormulario('updateReservacion');

        } catch (error) {
            console.error('Error al actualizar reservación:', error);
        }
    }
    ```

    **Endpoint**: `PUT /reservaciones/:id`

    **Request Body**:
    ```json
    {
      "folio": "RES-2024-001-UPDATED",
      "nombrecliente": "John M. Doe"
    }
    ```

    The selected reservation is stored in `reservacionSeleccionada` for the update operation.
  </Tab>

  <Tab title="Delete Reservation">
    ```javascript
    async function eliminarReservacion(id) {
        try {
            const response = await fetch(`${URL_RESERVACIONES}/${id}`, { 
                method: "DELETE" 
            });
            
            if (!response.ok) throw new Error('Error al eliminar reservación');

            loadReservaciones();

        } catch (error) {
            console.error('Error al eliminar reservación:', error);
        }
    }
    ```

    **Endpoint**: `DELETE /reservaciones/:id`

    The reservation list automatically refreshes after deletion.
  </Tab>
</Tabs>

## Helper Functions

### Form Cleanup

The system includes a utility function to reset forms and close modals:

```javascript
function limpiarFormulario(modalId) {
    document.getElementById(modalId).querySelectorAll("input").forEach(input => input.value = '');
    const modal = bootstrap.Modal.getInstance(document.getElementById(modalId));
    modal.hide();
}
```

This ensures forms are clean for the next operation and properly closes Bootstrap modals.

## Reservation Workflow

  </Step>
  <Step title="Availability Check">
    Staff checks room availability for requested dates and room type
  </Step>

  </Step>
  <Step title="Confirmation">

    Upon arrival, staff updates reservation status to checked-in
  </Step>

## Data Validation

Implement these validations for robust reservation management:

<Accordion title="Required Fields">
  - Folio number must be unique
  - Client information must be complete
  - Check-in date cannot be in the past
  - Check-out date must be after check-in date
</Accordion>

<Accordion title="Availability Checks">
  - Selected room must be available for the entire stay
  - Prevent double-booking of rooms
  - Block rooms under maintenance from booking
</Accordion>

<Accordion title="Business Rules">
  - Minimum stay requirements
  - Maximum advance booking period
  - Cancellation policy enforcement
  - Deposit requirements based on length of stay
</Accordion>
