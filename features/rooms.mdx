## Overview

The Room Management module allows hotel staff to maintain the complete room inventory. This includes adding new rooms, updating room details, managing pricing, and categorizing rooms by type.

<Note>
  All room operations interact with the API endpoint `http://localhost:4000/habitaciones`
</Note>

## User Interface

The room management interface (`habitaciones.html`) provides a table view displaying:

- **ID (#)**: Unique room identifier
- **Numero**: Room number or name
- **Tipo**: Room type (e.g., Single, Double, Suite)
- **Precio**: Room price per night
- **Acciones**: Action buttons for updates and deletions

## Managing Rooms

<Steps>
  <Step title="View Room Inventory">
    The room list automatically loads when accessing the page. All rooms are displayed in a sortable table format with complete details.

    ```javascript
    async function loadDataHabitacion() {
        const response = await fetch(URL);
        data = await response.json();
        
        // Renders each room into the table
        data.forEach(habitacion => {
            // Display room information
        });
    }
    ```

    The system fetches all room records from the backend and dynamically populates the table.
  </Step>

  <Step title="Add a New Room">
    Click the **"Nueva Habitación"** (New Room) button to open the room creation modal.

    <Accordion title="Room Creation Form Fields">
      The form requires three pieces of information:

      **Numero** (Number)
      - The room identifier or number (e.g., "101", "Penthouse Suite")
      - This is how staff and guests will reference the room

      **Tipo** (Type)
      - Room category (e.g., "Single", "Double", "Suite", "Deluxe")
      - Used for filtering and categorization

      **Precio** (Price)
      - Nightly rate for the room
      - Should be entered as a numeric value

      After completing all fields, click **"Guardar"** (Save) to add the room to inventory.
    </Accordion>
  </Step>

  <Step title="Update Room Details">
    To modify existing room information, click the **"Actualizar"** (Update) button next to the room.

    A dialog opens with pre-populated fields showing current room details. You can update:
    - Room number/name
    - Room type
    - Pricing

    Click **"Actualizar"** to save changes or **"Cerrar"** (Close) to cancel without saving.
  </Step>

  <Step title="Remove a Room">
    Click the **"Eliminar"** (Delete) button to remove a room from the inventory.

    <Note>
      Ensure the room has no active or future reservations before deletion. The system should validate this server-side.
    </Note>
  </Step>
</Steps>

## API Implementation

The room management system implements full CRUD functionality through RESTful endpoints:

<Tabs>
  <Tab title="Create Room">
    ```javascript
    async function guardarDataHabitacion() {
        const dataHabitacion = {
            nombre: document.getElementById("roomName").value,
            tipo: document.getElementById("roomType").value,
            precio: document.getElementById("roomPrice").value
        };

        try {
            const response = await fetch(URL, {
                method: "POST",
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(dataHabitacion)
            });

            if (!response.ok) throw new Error('Error al guardar habitación');
            loadDataHabitacion();

        } catch (error) {
            console.error(error);
        }
    }
    ```

    **Endpoint**: `POST /habitaciones`

    **Request Body**:
    ```json
    {
      "nombre": "101",
      "tipo": "Double",
      "precio": "150.00"
    }
    ```

    After successful creation, the room list automatically refreshes to display the new room.
  </Tab>

  <Tab title="Read Rooms">
    ```javascript
    async function loadDataHabitacion() {
        try {
            const response = await fetch(URL);
            data = await response.json();

            let html = '';
            data.forEach(habitacion => {
                html += `
                <tr>
                    <td>${habitacion.id}</td>
                    <td>${habitacion.nombre}</td>
                    <td>${habitacion.tipo}</td>
                    <td>${habitacion.precio}</td>
                    <td>
                        <button class="btn btn-primary btn-sm" 
                                onclick="modificarDataHabitacion(${habitacion.id})">
                            Actualizar
                        </button>
                        <button class="btn btn-danger btn-sm" 
                                onclick="deleteDataHabitacion(${habitacion.id})">
                            Eliminar
                        </button>
                    </td>
                </tr>`;
            });

            T_DATA_HABITACION.innerHTML = html;

        } catch (error) {
            console.error(error);
        }
    }
    ```

    **Endpoint**: `GET /habitaciones`

    **Response**:
    ```json
    [
      {
        "id": 1,
        "nombre": "101",
        "tipo": "Double",
        "precio": "150.00"
      },
      {
        "id": 2,
        "nombre": "202",
        "tipo": "Suite",
        "precio": "300.00"
      }
    ]
    ```
  </Tab>

  <Tab title="Update Room">
    ```javascript
    async function actualizarDataHabitacion() {
        const dialog = document.getElementById("formDataHabitacion");
        const id = dialog.dataset.habitacionId;

        const dataHabitacion = {
            nombre: document.getElementById("roomNameU").value,
            tipo: document.getElementById("roomTypeU").value,
            precio: document.getElementById("roomPriceU").value
        };

        try {
            const response = await fetch(`${URL}/${id}`, {
                method: "PUT",
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(dataHabitacion)
            });

            if (!response.ok) throw new Error('Error al actualizar habitación');

            dialog.close();
            loadDataHabitacion();

        } catch (error) {
            console.error(error);
        }
    }
    ```

    **Endpoint**: `PUT /habitaciones/:id`

    **Request Body**:
    ```json
    {
      "nombre": "101A",
      "tipo": "Deluxe Double",
      "precio": "175.00"
    }
    ```

    The dialog stores the room ID in its dataset attribute for the update operation.
  </Tab>

  <Tab title="Delete Room">
    ```javascript
    async function deleteDataHabitacion(id) {
        try {
            const response = await fetch(`${URL}/${id}`, { method: "DELETE" });
            if (!response.ok) throw new Error('Error al eliminar habitación');

            loadDataHabitacion();

        } catch (error) {
            console.error(error);
        }
    }
    ```

    **Endpoint**: `DELETE /habitaciones/:id`

    The room list automatically refreshes after successful deletion.
  </Tab>
</Tabs>

### Update Dialog

The update functionality uses a native HTML dialog element:

```javascript
function modificarDataHabitacion(id) {
    const dataHabitacion = data.find(habitacion => habitacion.id === id);
    document.getElementById("roomNameU").value = dataHabitacion.nombre;
    document.getElementById("roomTypeU").value = dataHabitacion.tipo;
    document.getElementById("roomPriceU").value = dataHabitacion.precio;

    const dialog = document.getElementById("formDataHabitacion");
    dialog.dataset.habitacionId = id;
    dialog.showModal();
}
```

The room ID is stored in the dialog's dataset for use during the update operation.
